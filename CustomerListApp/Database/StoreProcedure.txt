
CREATE PROCEDURE  SP_USERDETAILS
@USERNAME NVARCHAR(24),
@PASSWORD NVARCHAR(8),
@P_FlAG VARCHAR(1)
AS
BEGIN
IF(@P_FlAG ='S')
BEGIN
SELECT * FROM USERDETAILS WHERE USERNAME=@USERNAME AND PASSWORD=@PASSWORD
END
IF(@P_FlAG ='I')
BEGIN

INSERT INTO USERDETAILS(USERNAME,PASSWORD,OPENDT,CLOSEDT)
VALUES(@USERNAME,@PASSWORD,GETDATE(),NULL)

END
END
go
CREATE PROCEDURE  SP_CUSTOMERDETAILS
@CUSTOMERNAME VARCHAR(20),
@PHONENUMBER VARCHAR(12),
@ADDRESS VARCHAR(120),
@DATEOFBIRTH DATETIME,
@EMAILADDRESS VARCHAR(24)
AS
BEGIN

INSERT INTO CUSTOMERDETAILS(CUSTOMERNAME,PHONENUMBER,ADDRESS,DATEOFBIRTH,EMAIL,OPENDT,CLOSEDT)
VALUES(@CUSTOMERNAME,@PHONENUMBER,@ADDRESS,@DATEOFBIRTH,@EMAILADDRESS,GETDATE(),NULL)

END
go
Create  PROCEDURE  SP_CUSTOMERDETAILS_SELECT
AS
BEGIN

SELECT CUSTOMERNAME AS [Customer Name],PHONENUMBER As [Phone Number],ADDRESS As Address,FORMAT (DATEOFBIRTH, 'dd/MM/yyyy ') As [Date of Birth]  From CUSTOMERDETAILS

END
